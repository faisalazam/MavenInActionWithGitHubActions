15  @SpringBootTest

17  //Another useful approach is to not start the server at all but to test only the layer below that, where Spring handles
18  // the incoming HTTP request and hands it off to your controller. That way, almost of the full stack is used,
19  // and your code will be called in exactly the same way as if it were processing a real HTTP request
20  // but without the cost of starting the server. To do that, use Springâ€™s MockMvc and ask for that to be injected for
21  // you by using the @AutoConfigureMockMvc annotation on the test case.
22  class StartFullSpringContextButWithoutServerMockMvcHttpRequestIT {
23      @Autowired
24      private MockMvc mockMvc;
25  
26      @Test
27      public void shouldReturnDefaultMessage() throws Exception {
28          this.mockMvc.perform(get("/greeting")).andDo(print()).andExpect(status().isOk())
29                  .andExpect(content().string(containsString("World")));
30      }
31  }
